function getTrain() { var xhr = new XMLHttpRequest(); // Novo proxy mais rápido e compatível var target = 'https://v6.db.transport.rest/stops/8000105/departures?duration=60&results=10'; var url = 'https://corsproxy.io/?' + encodeURIComponent(target); xhr.open('GET', url, true); xhr.onreadystatechange = function() { if (xhr.readyState == 4) { if (xhr.status == 200) { try { var data = JSON.parse(xhr.responseText); var departures = data.departures; var html = ''; for (var i = 0; i < Math.min(departures.length, 6); i++) { var train = departures[i]; var name = train.line.name; var dest = train.direction.substring(0, 12).toUpperCase(); var bgColor = "#db0011"; if (train.line.productName === 'suburban') bgColor = "#008d3f"; if (name.indexOf('ICE') > -1) bgColor = "#ff0000"; var depTime = new Date(train.when || train.plannedWhen); var diff = Math.round((depTime - new Date()) / 60000); var tText = (diff <= 0) ? "JETZT" : diff + " min"; html += '
' + '' + name + '' + '' + dest + '' + '' + tText + '' + '
'; } document.getElementById('list').innerHTML = html || "KEINE ZÜGE"; } catch(e) { document.getElementById('list').innerHTML = "DATENFEHLER"; } } else { document.getElementById('list').innerHTML = "CONNECTION ERROR: " + xhr.status; } } }; xhr.send(); }
